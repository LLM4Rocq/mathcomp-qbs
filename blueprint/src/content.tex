\chapter*{Introduction}

Quasi-Borel spaces solve a fundamental problem: the category of
measurable spaces is not cartesian closed, which prevents giving
semantics to higher-order probabilistic programs. QBS replaces
measurable sets with ``random elements'' (morphisms from~$\mathbb{R}$),
yielding a cartesian-closed category with a well-behaved probability
monad.

This formalization follows:
\begin{itemize}
\item C.~Heunen, O.~Kammar, S.~Staton, H.~Yang.
  \emph{A Convenient Category for Higher-Order Probability Theory.}
  LICS 2017.
  \url{https://arxiv.org/abs/1701.02547}
\item M.~Hirata, Y.~Minamide, T.~Sato.
  \emph{Quasi-Borel Spaces.}
  Archive of Formal Proofs, 2022.
  \url{https://www.isa-afp.org/entries/Quasi_Borel_Spaces.html}
\end{itemize}

\chapter{Core QBS Theory}\label{ch:core}

In this chapter we define quasi-Borel spaces, their morphisms, and
the fundamental categorical constructions: products, exponentials,
coproducts, dependent products, and the unit type.
We establish cartesian closure, the $L \dashv R$ adjunction between
QBS and measurable spaces, and introduce standard Borel spaces
with their full faithfulness property.

The development is parametric in a \texttt{realType}~$R$ from math-comp
analysis. We write $\mathbb{R}$ for the carrier of~$R$ and
$m\mathbb{R}$ for its Borel measurable space.

\section{QBS Definition}

\begin{definition}[Quasi-Borel Space]\label{def:isQBS}
  \rocq{QBS.quasi_borel.isQBS}
  \rocqok
  A \emph{quasi-Borel space} (QBS) over $R$ is a type~$T$ equipped with
  a set $M_X \subseteq (\mathbb{R} \to T)$ of \emph{random elements}
  satisfying three axioms:
  \begin{enumerate}
    \item \textbf{Composition} (\texttt{qbs\_Mx\_comp}):
      If $\alpha \in M_X$ and $f : \mathbb{R} \to \mathbb{R}$ is
      measurable, then $\alpha \circ f \in M_X$.
    \item \textbf{Constants} (\texttt{qbs\_Mx\_const}):
      For every $x : T$, the constant function
      $\lambda r.\; x$ is in $M_X$.
    \item \textbf{Gluing} (\texttt{qbs\_Mx\_glue}):
      If $P : \mathbb{R} \to \mathbb{N}$ is measurable and
      $F_i \in M_X$ for all~$i$, then
      $\lambda r.\; F_{P(r)}(r) \in M_X$.
  \end{enumerate}
  The HB structure \texttt{QBSpace} packages this mixin with a short
  type name \texttt{qbsType R}.
\end{definition}


\section{Morphisms}\label{sec:morphisms}

\begin{definition}[QBS Morphism]\label{def:qbs_morphism}
  \rocq{QBS.quasi_borel.qbs_morphism}
  \rocqok
  \uses{def:isQBS}
  A function $f : X \to Y$ between QBS is a \emph{QBS morphism} if for
  every $\alpha \in M_X$, we have $f \circ \alpha \in M_Y$.

\begin{proof}\rocqok\end{proof}

\section{Exponentials (Function Spaces)}\label{sec:exponentials}

\begin{definition}[Exponential QBS]\label{def:expQ}
  \rocq{QBS.quasi_borel.expQ}
  \rocqok
  \uses{def:isQBS, def:qbs_morphism}
  The exponential $X \Rightarrow Y$ has carrier the set of QBS morphisms
  $X \to Y$, with a function $g : \mathbb{R} \to (X \to Y)$ random iff the
  uncurried map $(r, x) \mapsto g(r)(x)$ is a QBS morphism
  $\mathbb{R} \times X \to Y$.
\end{definition}

\begin{theorem}[Evaluation Morphism]\label{thm:qbs_morphism_eval}
  \rocq{QBS.quasi_borel.qbs_morphism_eval}
  \rocqok
  \uses{def:expQ, def:prodQ, def:qbs_morphism}
  Evaluation $(f, x) \mapsto f(x)$ is a QBS morphism
  $(X \Rightarrow Y) \times X \to Y$.
\end{theorem}
\begin{proof}\rocqok\end{proof}

\begin{theorem}[Currying Morphism]\label{thm:qbs_morphism_curry}
  \rocq{QBS.quasi_borel.qbs_morphism_curry}
  \rocqok
  \uses{def:expQ, def:prodQ, def:qbs_morphism, lem:prodQ_const_random}
  If $f : X \times Y \to Z$ is a QBS morphism, then
  $\hat{f} : X \to (Y \Rightarrow Z)$ defined by
  $\hat{f}(x)(y) = f(x,y)$ is a QBS morphism.
\end{theorem}
\begin{proof}\rocqok\end{proof}

\begin{corollary}[Cartesian Closure]\label{cor:cartesian_closure}
  \rocqok
  \uses{thm:qbs_morphism_eval, thm:qbs_morphism_curry}
  The category of quasi-Borel spaces is cartesian closed.
\end{corollary}
\begin{proof}\rocqok\end{proof}

\begin{lemma}[Exponential Composition]\label{lem:qbs_morphism_exp_comp}

\begin{theorem}[$L \dashv R$ Hom-Set Bijection]\label{thm:lr_adj_iff}
  \rocq{QBS.measure_qbs_adjunction.lr_adj_iff}
  \rocqok
  \uses{def:L_sigma, def:R_qbs, def:qbs_morphism}
  A function $f : X \to M$ is a QBS morphism $X \to R(M)$ if and only if
  it pulls back measurable sets to $L(X)$-measurable sets:
  \[
    \mathrm{Hom}_{\mathbf{QBS}}(X, R(M)) \cong
    \mathrm{Hom}_{\sigma\text{-}\mathbf{Alg}}(L(X), \sigma(M)).
  \]
  This is a single-object biconditional, not a full categorical
  naturality statement (the functorial squares are not formalized).
\end{theorem}
\begin{proof}

\chapter{Standard Borel Spaces ($\mathbb{R} \cong \mathbb{R} \times \mathbb{R}$)}\label{ch:standard_borel}

This chapter constructs the measurable bijection
$\mathbb{R} \cong \mathbb{R} \times \mathbb{R}$, proving that
the product of standard Borel spaces is standard Borel.
The construction composes three layers: (1) a bijection
$\mathbb{R} \leftrightarrow (0,1)$ via arctangent, (2) a bijection
$(0,1)^2 \leftrightarrow (0,1)$ via binary digit interleaving,
and (3) the composed encode/decode pair.

  \rocq{QBS.standard_borel.phi}

  \rocqok

  $\mathbb{R} \to (0,1)$.
  $\psi(y) = \tan(\pi(y - 1/2))$ is the inverse.
\end{definition}

\begin{lemma}[Phi Range]\label{lem:phi_range}

\begin{proof}\rocqok\end{proof}

\begin{lemma}[Phi--Psi Cancellation]\label{lem:phi_psi_cancel}

  \rocqok
  \uses{def:phi_psi}

\end{lemma}
\begin{proof}\rocqok\end{proof}

\section{Binary Digit Machinery}\label{sec:binary_digits}

\begin{definition}[Binary Digits and Sums]\label{def:bin_digits}
  \rocq{QBS.standard_borel.bin_digit}
  \rocq{QBS.standard_borel.bin_partial_sum}
  \rocq{QBS.standard_borel.bin_sum}
  \rocq{QBS.standard_borel.bin_digits}
  \rocqok
  $\mathrm{bin\_digit}(x, n)$ extracts the $n$-th binary digit of
  $x \in [0,1)$ via the doubling map.
  $\mathrm{bin\_partial\_sum}(d, n) = \sum_{i<n} d(i) \cdot 2^{-(i+1)}$
  and $\mathrm{bin\_sum}(d) = \lim_n \mathrm{bin\_partial\_sum}(d, n)$.
\end{definition}

\begin{lemma}[Convergence and Bounds]\label{lem:bin_sum_bounds}
  \rocq{QBS.standard_borel.is_cvg_bin_partial_sum}
  \rocq{QBS.standard_borel.bin_sum_le1}
  \rocq{QBS.standard_borel.bin_sum_ge0}
  \rocq{QBS.standard_borel.bin_partial_sum_nd}
  \rocqok
  \uses{def:bin_digits}
  Partial sums converge, are non-decreasing, and the limit satisfies
  $0 \leq \mathrm{bin\_sum}(d) \leq 1$.
\end{lemma}
\begin{proof}\rocqok\end{proof}

\begin{lemma}[Geometric Sum Identity]\label{lem:geom_half_sum}
  \rocq{QBS.standard_borel.geom_half_sum}
  \rocqok
  $\sum_{i<n} (1/2)^{i+1} = 1 - (1/2)^n$.
\end{lemma}
\begin{proof}\rocqok\end{proof}

\begin{theorem}[Binary Digit Reconstruction]\label{thm:bin_digits_reconstruction}
  \rocq{QBS.standard_borel.bin_digits_reconstruction}
  \rocqok
  \uses{def:bin_digits, lem:bin_sum_bounds}
  For $0 \leq x < 1$: $\mathrm{bin\_sum}(\mathrm{bin\_digits}(x)) = x$.
\end{theorem}
\begin{proof}
  \rocqok
  \uses{lem:bin_sum_bounds, lem:geom_half_sum}
  Show $x - \mathrm{bin\_partial\_sum}(n) = \mathrm{rem}(n) \cdot (1/2)^n$
  with $0 \leq \mathrm{rem}(n) < 1$, so the error converges to~$0$.
\end{proof}

  \rocq{QBS.standard_borel.no_trailing_ones}

\chapter{QBS--Kernel Bridge}\label{ch:qbs_kernel}

This chapter bridges the QBS probability monad with the s-finite
kernel infrastructure from \texttt{mathcomp-analysis} (the merged
kernel hierarchy~\texttt{kernel.v}, originally from PR~\#896).
S-finite kernels provide the standard semantic foundation for
first-order probabilistic programs (Staton, ESOP~2017): they are
closed under composition, unlike $\sigma$-finite kernels.
The bridge developed here shows that QBS probabilities induce
s-finite measures and that QBS morphisms between standard Borel
spaces lift to probability kernels, so kernel-based programs
embed faithfully into the QBS world while gaining higher-order
function types.
